00050         : vertices(vertices)

00052         , currentVertex(0)
00053     {
00054         Q_ASSERT_X(vertices.size() != 1 && vertices.size() != 2, Q_FUNC_INFO,
00055                    "A vertex array containing triangles of any primitive type must not "
00056                    "have 1 or 2 vertices");
00057         switch(primitiveType)
00058         {
00059             case PrimitiveType_Triangles:
00060                 Q_ASSERT_X(vertices.size() % 3 == 0, Q_FUNC_INFO,
00061                            "Size of a vertex array of triangles must be divisible by 3");
00062                 currentVertex = 0;
00063                 break;
00064             case PrimitiveType_TriangleStrip:
00065                 currentVertex = 2;
00066                 break;
00067             case PrimitiveType_TriangleFan:
00068                 currentVertex = 1;
00069                 break;
00070             default:
00071                 Q_ASSERT_X(false, Q_FUNC_INFO, "Not a triangle primitive type");
00072         }
00073     }

00083     bool next(V& a, V& b, V& c)
00084     {
00085         Q_ASSERT_X(currentVertex > 0, Q_FUNC_INFO, "Current vertex is negative");
00086         Q_ASSERT_X(currentVertex <= vertices.size(), Q_FUNC_INFO,
00087                    "Iterating outside the vertex array");
00088         if(currentVertex == vertices.size()){return false;}
00089 
00090         switch(primitiveType)
00091         {
00092             case PrimitiveType_Triangles:
00093                 a = vertices[currentVertex++];
00094                 b = vertices[currentVertex++];
00095                 c = vertices[currentVertex++];
00096                 break;
00097             case PrimitiveType_TriangleStrip:
00098                 if(currentVertex % 2 == 0)
00099                 {
00100                     a = vertices[currentVertex - 2];
00101                     b = vertices[currentVertex - 1];
00102                 }
00103                 else
00104                 {
00105                     a = vertices[currentVertex - 1];
00106                     b = vertices[currentVertex - 2];
00107                 }
00108                 c = vertices[currentVertex];
00109                 ++currentVertex;
00110                 break;
00111             case PrimitiveType_TriangleFan:
00112                 a = vertices[0];
00113                 b = vertices[currentVertex];
00114                 c = vertices[currentVertex + 1];
00115                 ++currentVertex;
00116                 break;
00117             default:
00118                 Q_ASSERT_X(false, Q_FUNC_INFO, "Not a triangle primitive type");
00119         }
00120 
00121         return true;
00122     }
