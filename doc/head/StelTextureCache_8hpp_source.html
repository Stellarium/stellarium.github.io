00053 class StelTextureCache

00057     struct TextureBackendRefCounted

00083     virtual TextureBackend* get(const QString& name, const TextureLoadingMode loadingMode)
00084     {
00085         Q_ASSERT_X(!name.isEmpty(), Q_FUNC_INFO, "Can't cache textures without a name");
00086 
00087         if(!cache.contains(name)) {return NULL;}
00088 
00089         TextureBackend* result = cache[name].backend;
00090 
00091         switch(loadingMode)
00092         {
00093             case TextureLoadingMode_Normal:
00094                 //If cached texture is loading (asynchronous), wait,
00095                 //if uninitialized(lazy), start loading and wait
00096                 if(result->getStatus() == TextureStatus_Uninitialized)
00097                 {
00098                     result->startAsynchronousLoading();
00099                 }
00100                 //Waiting in a hot loop, this could be done better
00101                 while(result->getStatus() == TextureStatus_Loading)
00102                 {
00103                     continue;
00104                 }
00105             case TextureLoadingMode_Asynchronous:
00106                 //If cached texture is uninitialized (lazy), start loading and wait.
00107                 if(result->getStatus() == TextureStatus_Uninitialized)
00108                 {
00109                     result->startAsynchronousLoading();
00110                 }
00111                 break;
00112             case TextureLoadingMode_LazyAsynchronous: 
00113                 break;
00114             default:
00115                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown texture loading mode");
00116         }
00117 
00118         (cache[name].refCount)++;
00119 
00120         return result;
00121     }
00122 
00129     virtual void add(TextureBackend* backend)
00130     {
00131         const QString& name = backend->getName();
00132         Q_ASSERT_X(!name.isEmpty(), Q_FUNC_INFO, "Can't cache textures without a name");
00133         Q_ASSERT_X(!cache.contains(name), Q_FUNC_INFO, 
00134                    "Adding the same texture to the cache more than once");
00135 
00136         cache.insert(name, TextureBackendRefCounted(backend));
00137     }
00138 
00145     virtual void remove(TextureBackend* backend)
00146     {
00147         const QString& name = backend->getName();
00148         Q_ASSERT_X(!name.isEmpty(), Q_FUNC_INFO, "Can't cache textures without a name");
00149         Q_ASSERT_X(cache.contains(name), Q_FUNC_INFO,
00150                    "Trying to remove unknown texture from cache");
00151 
00152         TextureBackendRefCounted& cached = cache[name];
00153         --(cached.refCount);
00154         Q_ASSERT_X(cached.refCount >= 0, Q_FUNC_INFO,
00155                    "Negative reference count of a texture in texture cache");
00156 
00157         // If reference count hits 0, delete the texture.
00158         if(cached.refCount == 0)
00159         {
00160             // Assert that we're really deleting what we thing we're deleting
00161             // instead of having two textures with the same name.
00162             Q_ASSERT_X(backend == cached.backend, Q_FUNC_INFO,
00163                        "Texture in cache has the same name as but is not specified texture");
00164             cache.remove(name);
00165             delete backend;
00166         }
00167     }
00168 
00169 private:
00171     QMap<QString, TextureBackendRefCounted> cache;
00172 };
