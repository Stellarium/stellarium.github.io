00137     void setVertexNonVirtual(const int index, const void* const vertexInPtr)
00138     {
00139         // Points to the current attribute (e.g. color, normal, vertex) within the vertex.
00140         const unsigned char* attribPtr = static_cast<const unsigned char*>(vertexInPtr);
00141         for(int attrib = 0; attrib < attributes.count; ++attrib)
00142         {
00143             const AttributeInterpretation interpretation = 
00144                 attributes.attributes[attrib].interpretation;
00145             //Set each attribute in its buffer.
00146             switch(attributes.attributes[attrib].type)
00147             {
00148                 case AttributeType_Vec2f:
00149                     getAttribute<Vec2f>(interpretation, index) 
00150                         = *reinterpret_cast<const Vec2f*>(attribPtr);
00151                     break;
00152                 case AttributeType_Vec3f:
00153                     getAttribute<Vec3f>(interpretation, index) 
00154                         = *reinterpret_cast<const Vec3f*>(attribPtr);
00155                     break;
00156                 case AttributeType_Vec4f:
00157                     getAttribute<Vec4f>(interpretation, index) 
00158                         = *reinterpret_cast<const Vec4f*>(attribPtr);
00159                     break;
00160                 default:
00161                     Q_ASSERT(false);
00162             }
00163 
00164             // This always works, because the C standard requires that 
00165             // sizeof(unsigned char) == 1  (that 1 might mean e.g. 16 bits instead of 8
00166             // on some platforms, but both the size of attribute and of unsigned char is 
00167             // measured in the same unit, so it's not a problem.
00168             attribPtr += attributes.sizes[attrib];
00169         }
00170     }
