00159     void setVertexNonVirtual(const int index, const void* const vertexInPtr)
00160     {
00161         // Points to the current attribute (e.g. color, normal, vertex) within the vertex.
00162         const unsigned char* attribPtr = static_cast<const unsigned char*>(vertexInPtr);
00163         for(int attrib = 0; attrib < attributes.count; ++attrib)
00164         {
00165             //Set each attribute in its buffer.
00166             switch(attributes.attributes[attrib].type)
00167             {
00168                 case AttributeType_Vec2f:
00169                     getAttribute<Vec2f>(attrib, index) = *reinterpret_cast<const Vec2f*>(attribPtr);
00170                     break;
00171                 case AttributeType_Vec3f:
00172                     getAttribute<Vec3f>(attrib, index) = *reinterpret_cast<const Vec3f*>(attribPtr);
00173                     break;
00174                 case AttributeType_Vec4f:
00175                     getAttribute<Vec4f>(attrib, index) = *reinterpret_cast<const Vec4f*>(attribPtr);
00176                     break;
00177                 default:
00178                     Q_ASSERT(false);
00179             }
00180 
00181             // This always works, because the C standard requires that 
00182             // sizeof(unsigned char) == 1  (that 1 might mean e.g. 16 bits instead of 8
00183             // on some platforms, but both the size of attribute and of unsigned char is 
00184             // measured in the same unit, so it's not a problem.
00185             attribPtr += attributes.sizes[attrib];
00186         }
00187     }
