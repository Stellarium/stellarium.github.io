00310     StelQGLGLSLShader* getShader(const StelVertexAttribute* const attributes, const int attributeCount)
00311     {
00312         invariant();
00313 
00314         // A custom shader (StelGLSLShader - in this case StelQGLGLSLShader)
00315         // overrides builtin shaders.
00316         if(NULL != customShader) {return customShader;}
00317 
00318         // Determine which vertex attributes are used.
00319         bool position, texCoord, normal, color;
00320         position = texCoord = normal = color = false;
00321         for(int attrib = 0; attrib < attributeCount; ++attrib)
00322         {
00323             const StelVertexAttribute& attribute(attributes[attrib]);
00324             switch(attribute.interpretation)
00325             {
00326                 case AttributeInterpretation_Position: position = true; break;
00327                 case AttributeInterpretation_TexCoord: texCoord = true; break;
00328                 case AttributeInterpretation_Normal:   normal   = true; break;
00329                 case AttributeInterpretation_Color:    color    = true; break;
00330                 default: 
00331                     Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown vertex interpretation");
00332             }
00333         }
00334 
00335         Q_ASSERT_X(position, Q_FUNC_INFO,
00336                    "Vertex formats without vertex position are not supported");
00337 
00338         invariant();
00339         // There are possible combinations - 4 are implemented right now.
00340         if(!texCoord && !normal && !color) {return plainShader;}
00341         if(!texCoord && !normal && color)  {return colorShader;}
00342         if(texCoord  && !normal && !color) {return textureShader;}
00343         if(texCoord  && !normal && color)  {return colorTextureShader;}
00344 
00345         // If we reach here, the vertex format has no shader implemented so we 
00346         // least inform the user about what vertex format fails (so it can
00347         // be changed or a shader can be implemented for it)
00348         qDebug() << "position: " << position << " texCoord: " << texCoord
00349                  << " normal: " << normal << " color: " << color;
00350 
00351         Q_ASSERT_X(false, Q_FUNC_INFO,
00352                    "Shader for vertex format not (yet) implemented");
00353 
00354         invariant();
00355         // Prevents GCC from complaining about exiting a non-void function:
00356         return NULL;
00357     }
