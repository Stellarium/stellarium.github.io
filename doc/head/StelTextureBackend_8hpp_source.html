00055 inline QString textureStatusName(const TextureStatus status)
00056 {
00057     switch(status)
00058     {
00059         case TextureStatus_Uninitialized: return "TextureStatus_Uninitialized"; break;
00060         case TextureStatus_Loaded:        return "TextureStatus_Loaded"; break;
00061         case TextureStatus_Loading:       return "TextureStatus_Loading"; break;
00062         case TextureStatus_Error:         return "TextureStatus_Error"; break;
00063         default: Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown texture status");
00064     }
00065     // Avoid compiler warnings.
00066     return QString();
00067 }
