69 template <class Star> class StarWrapper : public StarWrapperBase
70 {
71 protected:
72  StarWrapper(const SpecialZoneArray<Star> *a,
73  const SpecialZoneData<Star> *z,
74  const Star *s) : a(a), z(z), s(s) {;}
75  Vec3d getJ2000EquatorialPos(const StelCore* core) const
76  {
77  static const double d2000 = 2451545.0;
78  Vec3f v;
79  s->getJ2000Pos(z, (M_PI/180.)*(0.0001/3600.) * ((core->getJDay()-d2000)/365.25) / a->star_position_scale, v);
80  return Vec3d(v[0], v[1], v[2]);
81  }
82  Vec3f getInfoColor(void) const
83  {
84  return StelSkyDrawer::indexToColor(s->bV);
85  }
86  float getVMagnitude(const StelCore* core) const
87  {
88  Q_UNUSED(core);
89  return 0.001f*a->mag_min + s->mag*(0.001f*a->mag_range)/a->mag_steps;
90  }
91  float getBV(void) const {return s->getBV();}
92  QString getEnglishName(void) const {return QString();}
93  QString getNameI18n(void) const {return s->getNameI18n();}
94  virtual double getAngularSize(const StelCore*) const {return 0.;}
