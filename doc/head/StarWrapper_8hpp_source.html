76  Vec3d getJ2000EquatorialPos(const StelCore* core) const
77  {
78  static const double d2000 = 2451545.0;
79  Vec3f v;
80  s->getJ2000Pos(z, (M_PI/180.)*(0.0001/3600.) * ((core->getJDE()-d2000)/365.25) / a->star_position_scale, v);
81  return Vec3d(v[0], v[1], v[2]);
82  }
