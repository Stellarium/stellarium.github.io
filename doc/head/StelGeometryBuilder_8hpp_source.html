00574     void buildCircle(StelVertexBuffer<VertexP2>* vertexBuffer, 
00575                      const float x, const float y, const float radius, 
00576                      const int segments = 128)
00577     {
00578         Q_ASSERT_X(radius > 0.0f, Q_FUNC_INFO, "Circle must have a radius greater than zero");
00579         Q_ASSERT_X(segments > 3, Q_FUNC_INFO, "Circle must have at least 3 segments");
00580         Q_ASSERT_X(vertexBuffer->length() == 0, Q_FUNC_INFO,
00581                    "Need an empty vertex buffer to build a circle");
00582         Q_ASSERT_X(vertexBuffer->primitiveType() == PrimitiveType_LineStrip, Q_FUNC_INFO,
00583                    "Need a line loop vertex buffer to build a circle");
00584 
00585         const Vec2f center(x,y);
00586         const float phi = 2.0f * M_PI / segments;
00587         const float cp = std::cos(phi);
00588         const float sp = std::sin(phi);
00589         float dx = radius;
00590         float dy = 0;
00591 
00592         vertexBuffer->unlock();
00593 
00594         for (int i = 0; i < segments; i++)
00595         {
00596             vertexBuffer->addVertex(VertexP2(x + dx, y + dy));
00597             const float dxNew = dx * cp - dy * sp;
00598             dy = dx * sp + dy * cp;
00599             dx = dxNew;
00600         }
00601         vertexBuffer->addVertex(VertexP2(x + radius, y));
00602         vertexBuffer->lock();
00603     }

00614     void buildCylinder(StelVertexBuffer<VertexP3T2>* const vertexBuffer, 
00615                        const float radius, const float height, const int slices,
00616                        const bool orientInside = false)
00617     {
00618         // No need for indices - w/o top/bottom a cylinder is just a single tristrip
00619         // with no duplicate vertices except start/end
00620 
00621         Q_ASSERT_X(vertexBuffer->length() == 0, Q_FUNC_INFO, 
00622                    "Need an empty vertex buffer to start building a cylinder");
00623         Q_ASSERT_X(vertexBuffer->primitiveType() == PrimitiveType_TriangleStrip, 
00624                     Q_FUNC_INFO, "Need a triangle strip vertex buffer to build a cylinder");
00625         Q_ASSERT_X(slices >= 3, Q_FUNC_INFO, 
00626                    "can't build a cylinder with less than 3 slices");
00627 
00628         vertexBuffer->unlock();
00629         float texCoord = orientInside ? 1.0f : 0.0f;
00630         float angle    = 0.0f;
00631 
00632         // Needed to generate vertices with opposite winding if orientInside == true.
00633         const float sign          = orientInside ? -1.0f : 1.0f;
00634         const float deltaTexCoord = 1.0f / slices;
00635         const float deltaAngle    = 2.0f * M_PI / slices;
00636 
00637         for (int i = 0; i <= slices; ++i)
00638         {
00639             const float x = std::sin(angle) * radius;
00640             const float y = std::cos(angle) * radius;
00641 
00642             vertexBuffer->addVertex(VertexP3T2(Vec3f(x, y, 0.0f),   Vec2f(texCoord, 0.0f)));
00643             vertexBuffer->addVertex(VertexP3T2(Vec3f(x, y, height), Vec2f(texCoord, 1.0f)));
00644 
00645             // If oriented outside, we go from 0 at 0deg to 1 at 360deg
00646             // If oriented inside, we go from 1 at 360 deg to 0 at 0deg
00647             texCoord += sign * deltaTexCoord;
00648             angle    += sign * deltaAngle;
00649         }
00650         vertexBuffer->lock();
00651     }
