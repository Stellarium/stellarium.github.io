00358     void setupGLState(StelProjector* projector)
00359     {
00360         // Instead of setting GL state when functions such as setDepthTest() or setCulledFaces()
00361         // are called, we only set it before drawing and reset after drawing to avoid 
00362         // conflicts with e.g. Qt OpenGL backend, or any other GL code that might be running.
00363 
00364         // GL setup before drawing.
00365         // Fix some problem when using Qt OpenGL2 engine
00366         glStencilMask(0x11111111);
00367         
00368         // Moving this to drawVertexBufferBackend causes flicker issues
00369         switch(blendMode)
00370         {
00371             case BlendMode_None:
00372                 glDisable(GL_BLEND);
00373                 break;
00374             case BlendMode_Add:
00375                 glBlendFunc(GL_ONE, GL_ONE);
00376                 glEnable(GL_BLEND);
00377                 break;
00378             case BlendMode_Alpha:
00379                 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
00380                 glEnable(GL_BLEND);
00381                 break;
00382             case BlendMode_Multiply:
00383                 glBlendFunc(GL_SRC_COLOR, GL_ONE_MINUS_SRC_COLOR);
00384                 glEnable(GL_BLEND);
00385                 break;
00386             default:
00387                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown blend mode");
00388         }
00389 
00390         switch(depthTest)
00391         {
00392             case DepthTest_Disabled:
00393                 glDisable(GL_DEPTH_TEST);
00394                 break;
00395             case DepthTest_ReadOnly:
00396                 glEnable(GL_DEPTH_TEST);
00397                 glDepthMask(GL_FALSE);
00398                 break;
00399             case DepthTest_ReadWrite:
00400                 glEnable(GL_DEPTH_TEST);
00401                 glDepthMask(GL_TRUE);
00402                 break;
00403             default:
00404                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown depth test mode");
00405         }
00406 
00407         switch(stencilTest)
00408         {
00409             case StencilTest_Disabled:
00410                 glDisable(GL_STENCIL_TEST);
00411                 break;
00412             case StencilTest_Write_1:
00413                 glStencilFunc(GL_ALWAYS, 0x1, 0x1);
00414                 glStencilOp(GL_ZERO, GL_REPLACE, GL_REPLACE);
00415                 glEnable(GL_STENCIL_TEST);
00416                 break;
00417             case StencilTest_DrawIf_1:
00418                 glEnable(GL_STENCIL_TEST);
00419                 glStencilFunc(GL_EQUAL, 0x1, 0x1);
00420                 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
00421                 break;
00422             default:
00423                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown stencil test mode");
00424         }
00425 
00426         switch(culledFaces)
00427         {
00428             case CullFace_None:
00429                 glDisable(GL_CULL_FACE);
00430                 break;
00431             case CullFace_Front:
00432                 glEnable(GL_CULL_FACE);
00433                 glCullFace(GL_FRONT);
00434                 break;
00435             case CullFace_Back:
00436                 glEnable(GL_CULL_FACE);
00437                 glCullFace(GL_BACK);
00438                 break;
00439             default: Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown cull face type");
00440         }
00441         
00442         glFrontFace(projector->flipFrontBackFace() ? GL_CW : GL_CCW);
00443     }
