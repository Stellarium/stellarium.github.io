00392     void setupGLState(StelProjector* projector)
00393     {
00394         // Instead of setting GL state when functions such as setDepthTest() or setCulledFaces()
00395         // are called, we only set it before drawing and reset after drawing to avoid 
00396         // conflicts with e.g. Qt OpenGL backend, or any other GL code that might be running.
00397         //
00398         // Optimization note: 
00399         // Disabling GL state setup and reset improves performance by 3-8% 
00400         // (open source AMD driver), so that is the maximum speedup achievable 
00401         // by moving state setup into functions such as setBlendMode, setDepthTest, etc.
00402 
00403         // GL setup before drawing.
00404         // Fix some problem when using Qt OpenGL2 engine
00405         glStencilMask(0x11111111);
00406         
00407         // Moving this to drawVertexBufferBackend causes flicker issues
00408         switch(blendMode)
00409         {
00410             case BlendMode_None:
00411                 glDisable(GL_BLEND);
00412                 break;
00413             case BlendMode_Add:
00414                 glBlendFunc(GL_ONE, GL_ONE);
00415                 glEnable(GL_BLEND);
00416                 break;
00417             case BlendMode_Alpha:
00418                 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
00419                 glEnable(GL_BLEND);
00420                 break;
00421             case BlendMode_Multiply:
00422                 glBlendFunc(GL_SRC_COLOR, GL_ONE_MINUS_SRC_COLOR);
00423                 glEnable(GL_BLEND);
00424                 break;
00425             default:
00426                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown blend mode");
00427         }
00428 
00429         switch(depthTest)
00430         {
00431             case DepthTest_Disabled:
00432                 glDisable(GL_DEPTH_TEST);
00433                 break;
00434             case DepthTest_ReadOnly:
00435                 glEnable(GL_DEPTH_TEST);
00436                 glDepthMask(GL_FALSE);
00437                 break;
00438             case DepthTest_ReadWrite:
00439                 glEnable(GL_DEPTH_TEST);
00440                 glDepthMask(GL_TRUE);
00441                 break;
00442             default:
00443                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown depth test mode");
00444         }
00445 
00446         switch(stencilTest)
00447         {
00448             case StencilTest_Disabled:
00449                 glDisable(GL_STENCIL_TEST);
00450                 break;
00451             case StencilTest_Write_1:
00452                 glStencilFunc(GL_ALWAYS, 0x1, 0x1);
00453                 glStencilOp(GL_ZERO, GL_REPLACE, GL_REPLACE);
00454                 glEnable(GL_STENCIL_TEST);
00455                 break;
00456             case StencilTest_DrawIf_1:
00457                 glEnable(GL_STENCIL_TEST);
00458                 glStencilFunc(GL_EQUAL, 0x1, 0x1);
00459                 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
00460                 break;
00461             default:
00462                 Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown stencil test mode");
00463         }
00464 
00465         switch(culledFaces)
00466         {
00467             case CullFace_None:
00468                 glDisable(GL_CULL_FACE);
00469                 break;
00470             case CullFace_Front:
00471                 glEnable(GL_CULL_FACE);
00472                 glCullFace(GL_FRONT);
00473                 break;
00474             case CullFace_Back:
00475                 glEnable(GL_CULL_FACE);
00476                 glCullFace(GL_BACK);
00477                 break;
00478             default: Q_ASSERT_X(false, Q_FUNC_INFO, "Unknown cull face type");
00479         }
00480         
00481         glFrontFace(projector->flipFrontBackFace() ? GL_CW : GL_CCW);
00482     }

00553     struct FrameTimeQueue
00554     {
00555     public:
00557         FrameTimeQueue()
00558             : nextFrame(0)
00559         {
00560             memset(frames, '\0', FRAME_CAPACITY * sizeof(long double));

00566         void update()
00567         {
00568             frames[nextFrame++] = StelUtils::secondsSinceStart();
00569             nextFrame = nextFrame % FRAME_CAPACITY;
00570         }
00571 
00573         double getFPS()
00574         {
00575             const long double newest = frames[(nextFrame - 1 + 256) % FRAME_CAPACITY];
00576             long double oldest = frames[nextFrame];
00577             Q_ASSERT_X(newest >= oldest, Q_FUNC_INFO, 
00578                           "Oldest frame time is more recent than newest");
00579             int tooOldCount = 0;
00580             // Don't compute FPS from more than one second of frames.
00581             while(newest - oldest > 1.0 && tooOldCount < FRAME_CAPACITY - 1)
00582             {
00583                 ++tooOldCount;
00584                 oldest = frames[(nextFrame + tooOldCount) % FRAME_CAPACITY];
00585             }
00586             return (FRAME_CAPACITY - tooOldCount) / (newest - oldest);
00587         }
