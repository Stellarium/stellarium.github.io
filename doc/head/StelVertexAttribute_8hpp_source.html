00174     static void validateAttributes(const QVector<StelVertexAttribute>& attributes)
00175     {
00176         bool vertex, texCoord, normal, color;
00177         vertex = texCoord = normal = color = false;
00178         
00179         // Ensure that every kind of vertex attribute is present at most once.
00180         foreach(const StelVertexAttribute& attribute, attributes)
00181         {
00182             switch(attribute.interpretation)
00183             {
00184                 case AttributeInterpretation_Position:
00185                     Q_ASSERT_X(!vertex, Q_FUNC_INFO,
00186                                "Vertex type has more than one vertex position attribute");
00187                     vertex = true;
00188                     break;
00189                 case AttributeInterpretation_TexCoord:
00190                     Q_ASSERT_X(attributeDimensions(attribute.type) == 2, Q_FUNC_INFO,
00191                                "Only 2D texture coordinates are supported at the moment");
00192                     Q_ASSERT_X(!texCoord, 
00193                                Q_FUNC_INFO,
00194                                "Vertex type has more than one texture coordinate attribute");
00195                     texCoord = true;
00196                     break;
00197                 case AttributeInterpretation_Normal:
00198                     Q_ASSERT_X(attributeDimensions(attribute.type) == 3, Q_FUNC_INFO,
00199                                "Only 3D vertex normals are supported");
00200                     Q_ASSERT_X(!normal, Q_FUNC_INFO,
00201                                "Vertex type has more than one normal attribute");
00202                     normal = true;
00203                     break;
00204                 case AttributeInterpretation_Color:
00205                     Q_ASSERT_X(!color, Q_FUNC_INFO,
00206                                "Vertex type has more than one color attribute");
00207                     color = true;
00208                     break;
00209                 default:
00210                     Q_ASSERT_X(false, Q_FUNC_INFO,
00211                                "Unknown vertex attribute interpretation");
00212             }
00213         }
00214 
00215         Q_ASSERT_X(vertex, Q_FUNC_INFO, 
00216                    "Vertex formats without a position attribute are not supported");
00217     }
