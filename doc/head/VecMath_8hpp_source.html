00977 /*
00978  * Code ripped from the GLU library
00979  * Compute inverse of 4x4 transformation matrix.
00980  * Code contributed by Jacques Leroy jle@star.be
00981  * Return zero matrix on failure (singular matrix)
00982  */
00983 template<class T> Matrix4<T> Matrix4<T>::inverse() const
00984 {
00985     const T * m = r;
00986     T out[16];
00987 
00988 /* NB. Matrices used by StelRenderer are COLUMN major. */
00989 #define SWAP_ROWS(a, b) { T *_tmp = a; (a)=(b); (b)=_tmp; }
00990 #define MAT(m,r,c) (m)[(c)*4+(r)]
00991 
00992    T wtmp[4][8];
00993    T m0, m1, m2, m3, s;
00994    T *r0, *r1, *r2, *r3;
00995 
00996    r0 = wtmp[0], r1 = wtmp[1], r2 = wtmp[2], r3 = wtmp[3];
00997 
00998    r0[0] = MAT(m, 0, 0), r0[1] = MAT(m, 0, 1),
00999       r0[2] = MAT(m, 0, 2), r0[3] = MAT(m, 0, 3),
01000       r0[4] = 1.0, r0[5] = r0[6] = r0[7] = 0.0,
01001       r1[0] = MAT(m, 1, 0), r1[1] = MAT(m, 1, 1),
01002       r1[2] = MAT(m, 1, 2), r1[3] = MAT(m, 1, 3),
01003       r1[5] = 1.0, r1[4] = r1[6] = r1[7] = 0.0,
01004       r2[0] = MAT(m, 2, 0), r2[1] = MAT(m, 2, 1),
01005       r2[2] = MAT(m, 2, 2), r2[3] = MAT(m, 2, 3),
01006       r2[6] = 1.0, r2[4] = r2[5] = r2[7] = 0.0,
01007       r3[0] = MAT(m, 3, 0), r3[1] = MAT(m, 3, 1),
01008       r3[2] = MAT(m, 3, 2), r3[3] = MAT(m, 3, 3),
01009       r3[7] = 1.0, r3[4] = r3[5] = r3[6] = 0.0;
01010 
01011    /* choose pivot - or die */
01012    if (fabs(r3[0]) > fabs(r2[0]))
01013       SWAP_ROWS(r3, r2);
01014    if (fabs(r2[0]) > fabs(r1[0]))
01015       SWAP_ROWS(r2, r1);
01016    if (fabs(r1[0]) > fabs(r0[0]))
01017       SWAP_ROWS(r1, r0);
01018    if (0.0 == r0[0])
01019       return Matrix4<T>();
01020 
01021    /* eliminate first variable     */
01022    m1 = r1[0] / r0[0];
01023    m2 = r2[0] / r0[0];
01024    m3 = r3[0] / r0[0];
01025    s = r0[1];
01026    r1[1] -= m1 * s;
01027    r2[1] -= m2 * s;
01028    r3[1] -= m3 * s;
01029    s = r0[2];
01030    r1[2] -= m1 * s;
01031    r2[2] -= m2 * s;
01032    r3[2] -= m3 * s;
01033    s = r0[3];
01034    r1[3] -= m1 * s;
01035    r2[3] -= m2 * s;
01036    r3[3] -= m3 * s;
01037    s = r0[4];
01038    if (s != 0.0) {
01039       r1[4] -= m1 * s;
01040       r2[4] -= m2 * s;
01041       r3[4] -= m3 * s;
01042    }
01043    s = r0[5];
01044    if (s != 0.0) {
01045       r1[5] -= m1 * s;
01046       r2[5] -= m2 * s;
01047       r3[5] -= m3 * s;
01048    }
01049    s = r0[6];
01050    if (s != 0.0) {
01051       r1[6] -= m1 * s;
01052       r2[6] -= m2 * s;
01053       r3[6] -= m3 * s;
01054    }
01055    s = r0[7];
01056    if (s != 0.0) {
01057       r1[7] -= m1 * s;
01058       r2[7] -= m2 * s;
01059       r3[7] -= m3 * s;
01060    }
01061 
01062    /* choose pivot - or die */
01063    if (fabs(r3[1]) > fabs(r2[1]))
01064       SWAP_ROWS(r3, r2);
01065    if (fabs(r2[1]) > fabs(r1[1]))
01066       SWAP_ROWS(r2, r1);
01067    if (0.0 == r1[1])
01068       return Matrix4<T>();
01069 
01070    /* eliminate second variable */
01071    m2 = r2[1] / r1[1];
01072    m3 = r3[1] / r1[1];
01073    r2[2] -= m2 * r1[2];
01074    r3[2] -= m3 * r1[2];
01075    r2[3] -= m2 * r1[3];
01076    r3[3] -= m3 * r1[3];
01077    s = r1[4];
01078    if (0.0 != s) {
01079       r2[4] -= m2 * s;
01080       r3[4] -= m3 * s;
01081    }
01082    s = r1[5];
01083    if (0.0 != s) {
01084       r2[5] -= m2 * s;
01085       r3[5] -= m3 * s;
01086    }
01087    s = r1[6];
01088    if (0.0 != s) {
01089       r2[6] -= m2 * s;
01090       r3[6] -= m3 * s;
01091    }
01092    s = r1[7];
01093    if (0.0 != s) {
01094       r2[7] -= m2 * s;
01095       r3[7] -= m3 * s;
01096    }
01097 
01098    /* choose pivot - or die */
01099    if (fabs(r3[2]) > fabs(r2[2]))
01100       SWAP_ROWS(r3, r2);
01101    if (0.0 == r2[2])
01102       return Matrix4<T>();
01103 
01104    /* eliminate third variable */
01105    m3 = r3[2] / r2[2];
01106    r3[3] -= m3 * r2[3], r3[4] -= m3 * r2[4],
01107       r3[5] -= m3 * r2[5], r3[6] -= m3 * r2[6], r3[7] -= m3 * r2[7];
01108 
01109    /* last check */
01110    if (0.0 == r3[3])
01111       return Matrix4<T>();
01112 
01113    s = 1.0 / r3[3];     /* now back substitute row 3 */
01114    r3[4] *= s;
01115    r3[5] *= s;
01116    r3[6] *= s;
01117    r3[7] *= s;
01118 
01119    m2 = r2[3];          /* now back substitute row 2 */
01120    s = 1.0 / r2[2];
01121    r2[4] = s * (r2[4] - r3[4] * m2), r2[5] = s * (r2[5] - r3[5] * m2),
01122       r2[6] = s * (r2[6] - r3[6] * m2), r2[7] = s * (r2[7] - r3[7] * m2);
