95  c = vertices[currentVertex++];

97  case PrimitiveType_TriangleStrip:
98  if(currentVertex % 2 == 0)

100  a = vertices[currentVertex - 2];
101  b = vertices[currentVertex - 1];
102  }
103  else

107  }
108  c = vertices[currentVertex];
