78  {
79  static const double d2000 = 2451545.0;
80  Vec3f v;
81  s->getJ2000Pos(z, (M_PI/180.)*(0.0001/3600.) * ((core->getJDE()-d2000)/365.25) / a->star_position_scale, v);
