258 private:
259  bool operator < (const QTreeWidgetItem &other) const
260  {
261  int column = treeWidget()->sortColumn();
262 
263  if (column == AstroCalcDialog::CColumnName)
264  {
265  QRegExp dso("^(\\w+)\\s*(\\d+)\\s*(.*)$");
266  int a = 0, b = 0;
267  if (dso.exactMatch(text(column)))
268  a = dso.capturedTexts().at(2).toInt();
269  if (dso.exactMatch(other.text(column)))
270  b = dso.capturedTexts().at(2).toInt();
271  if (a>0 && b>0)
272  return a < b;
273  else
274  return text(column).toLower() < other.text(column).toLower();
275  }
276  else if (column == AstroCalcDialog::CColumnRA || column == AstroCalcDialog::CColumnDec)
277  {
278  return StelUtils::getDecAngle(text(column)) < StelUtils::getDecAngle(other.text(column));
279  }
280  else if (column == AstroCalcDialog::CColumnMagnitude || column == AstroCalcDialog::CColumnExtra)
281  {
282  return text(column).toFloat() < other.text(column).toFloat();
283  }
284  else
285  {
286  return text(column).toLower() < other.text(column).toLower();
287  }
288  }
